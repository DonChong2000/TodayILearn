function flood(image, sr, sc, color, target=null) {
    if target==null{
    //getsrsc color, set as target
    }
    
    If image[sr,sc] == target{
        image[sr,sc] = color;
        image =  flood(image,sr+1,sc,color)
        image = flood(image,sr-1,sc,color)
        image = flood(image,sr,sc+1,color)
        image = flood(image,sr,sc-1,color)
    }

    return image;
}
