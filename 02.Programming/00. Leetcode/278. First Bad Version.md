```js 
var solution = function(isBadVersion) {
    return function(n) {

        let head = 1;
        let tail = n;
        if (tail==1){return tail;}

        while (head<tail){
            let mid = Math.floor((head+tail)/2);
            if(isBadVersion(mid)){
                tail = mid;
            }else{
                head = mid;
                if(isBadVersion(mid+1)){
                    return mid+1;
                }
            }
        }
        return 1;
    };
};
```

```js
    /**
     * @param {integer} n Total versions
     * @return {integer} The first bad version
     */
    return function(n) {
        let left = 1;
        let right = n;
        
        while (left < right) {
            let mid = left + Math.floor((right - left) / 2);
            
            if (isBadVersion(mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        
        return left;
    };
};
```
