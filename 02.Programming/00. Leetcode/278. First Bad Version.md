var solution = function(isBadVersion) {
    return function(n) {

        let head = 1;
        let tail = n;
        if (tail==1){return tail;}

        while (head<tail){
            let mid = Math.floor((head+tail)/2);
            if(isBadVersion(mid)){
                tail = mid;
            }else{
                head = mid;
                if(isBadVersion(mid+1)){
                    return mid+1;
                }
            }
        }
        return 1;
    };
};
