var isPalindrome = function(s) {

    //https://stackoverflow.com/questions/38512374/quick-way-to-extract-alphanumeric-characters-from-a-string-in-javascript

    for (let i=0;i<mid;i++){
        if(s[i] != s[s.length-i-1]){
            return false;
        }
    }
return true;
};
