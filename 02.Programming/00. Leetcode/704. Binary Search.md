   if (left > right) return -1;
   let mid = Math.floor((left + right)/2);
   if(nums[mid] === target) return mid;
   else if (nums[mid] > target) return search(nums, target, left, mid - 1);
   else return search(nums, target, mid + 1, right);  
};
